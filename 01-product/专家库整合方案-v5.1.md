## 2. 与蓝军评审整合 (核心)

### 2.1 动态专家分配

**现状**: 蓝军评审固定4位专家（事实核查员、逻辑检察官、行业专家、读者代表）

**升级方案**:
```typescript
// 评审任务创建时自动匹配专家
interface ReviewAssignment {
  // 保留通用专家
  universalExperts: {
    factChecker: Expert;      // 事实核查员 - 检查数据准确性
    logicChecker: Expert;     // 逻辑检察官 - 检查论证严密性
    readerRep: Expert;        // 读者代表 - 检查可读性
  };

  // 新增：动态匹配的领域专家 (2-3位)
  domainExperts: Expert[];    // 基于任务主题匹配

  // 新增：可选特级专家 (重大任务)
  seniorExpert?: Expert;      // S级专家，用于战略评审
}

// 匹配算法
function assignExperts(task: Task): ReviewAssignment {
  // 1. 解析任务主题
  const domain = extractDomain(task.topic);

  // 2. 从对应领域匹配专家
  const domainExperts = expertLibrary
    .findByDomain(domain)
    .sortByCapability('review')
    .take(2);

  // 3. 判断是否启用特级专家
  const seniorExpert = task.importance > 0.8
    ? expertLibrary.findSeniorExpert(domain)
    : undefined;

  return {
    universalExperts: getUniversalExperts(),
    domainExperts,
    seniorExpert
  };
}
```

```typescript
interface ExpertReviewGenerator {
  // 生成专家观点
  generateReview(expert: Expert, content: string): ReviewOpinion {
    // 1. 加载专家核心思想
    const philosophy = expert.philosophy;

    // 2. 分析内容与专家思想的契合度
    const alignment = analyzeAlignment(content, philosophy);

    // 3. 基于专家风格生成观点
    return {
      expertId: expert.id,
      expertName: expert.name,

      // 观点内容 - 体现专家特色

      // 核心关注点 - 基于专家评审维度

      // 建议 - 基于专家成功实践
      suggestions: generateSuggestions(expert.achievements, content),

      // 置信度

      // 差异化标签
      differentiationTags: generateTags(expert, content)
    };
  }
}
```

### 3.3 专家工作量管理

```typescript
// 新增专家调度系统
interface ExpertScheduler {
  // 查看专家工作负载
  getExpertWorkload(expertId: string): {
    pendingReviews: number;
    avgReviewTime: number;
    availability: 'available' | 'busy' | 'unavailable';
  };

  // 智能分配 - 考虑专家负载
  assignWithLoadBalancing(task: Task): Expert[] {
    const candidates = expertLibrary.findByDomain(task.domain);
    return candidates
      .filter(e => this.getExpertWorkload(e.id).availability !== 'unavailable')
      .sortBy(e => this.getExpertWorkload(e.id).pendingReviews)
      .take(2);
  }
}
```

### 5.2 热点推荐算法增强

```typescript
// 结合专家思想的热点推荐
interface ExpertEnhancedRecommendation {
  // 原有推荐
  baseRecommendations: HotTopic[];

  // 专家增强
  expertBoost: {
    expert: Expert;
    reason: string;  // 为什么这个专家关注这个话题
    relevanceScore: number;
  }[];

  // 综合评分
  finalScore: number;
}

function calculateExpertBoost(
  topic: HotTopic,
  userInterests: UserProfile,
  experts: Expert[]
): ExpertEnhancedRecommendation {
  // 找到最关注这个话题的专家
  const relevantExperts = experts
    .filter(e => isExpertRelevant(e, topic))
    .map(e => ({
      expert: e,
      reason: generateRelevanceReason(e, topic),
      relevanceScore: calculateRelevance(e, topic)
    }));

  return {
    baseRecommendations: [topic],
    expertBoost: relevantExperts,
    finalScore: combineScores(topic.score, relevantExperts)
  };
}
```

```typescript
// 专家参与的可信度评估
interface ExpertCredibilityAssessment {
  // 基础评分
  baseScore: number;

  // 专家评审
  expertReviews: {
    expert: Expert;
    score: number;
    comments: string;
  }[];

  // 综合可信度
  finalCredibility: number;
}

function assessCredibilityWithExperts(
  asset: Asset,
  experts: Expert[]
): ExpertCredibilityAssessment {
  const expertReviews = experts.map(expert => ({
    expert,
    score: expert.evaluateCredibility(asset),
    comments: expert.generateCredibilityComment(asset)
  }));

  return {
    baseScore: calculateBaseScore(asset),
    expertReviews,
    finalCredibility: weightedAverage(expertReviews)
  };
}
```

## 8. 数据库设计

### 8.1 专家表

```sql
-- experts 表
CREATE TABLE experts (
  id VARCHAR(20) PRIMARY KEY,  -- E01-01, S-01
  name VARCHAR(100) NOT NULL,
  level VARCHAR(10),           -- 'domain', 'senior'
  domain_code VARCHAR(10),     -- E01, E02...
  domain_name VARCHAR(50),

  -- 人设信息 (JSON)
  profile JSONB,

  -- 核心思想 (JSON)
  philosophy JSONB,

  -- 成功实践 (JSON数组)
  achievements JSONB,

  -- 评审维度 (JSON数组)
  review_dimensions JSONB,

  -- 状态
  status VARCHAR(20) DEFAULT 'active',

  -- 统计
  total_reviews INT DEFAULT 0,
  acceptance_rate DECIMAL(4,2),
  avg_response_time INT,  -- 分钟

  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- 专家评审记录表
CREATE TABLE expert_reviews (
  id SERIAL PRIMARY KEY,
  expert_id VARCHAR(20) REFERENCES experts(id),
  task_id VARCHAR(50),
  content_type VARCHAR(20),

  -- 评审内容
  opinion TEXT,
  focus_areas JSONB,
  suggestions JSONB,
  confidence DECIMAL(3,2),
  differentiation_tags JSONB,

  -- 用户反馈
  user_action VARCHAR(20),  -- accept, ignore, partial
  user_note TEXT,

  created_at TIMESTAMP DEFAULT NOW()
);
```
