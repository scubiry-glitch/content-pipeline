## 2. 多租户数据库设计

### 2.1 Schema 隔离策略

采用 **Schema-per-Tenant** 方案：
- 每个租户拥有独立的 PostgreSQL schema
- 共享 public schema 存储租户元数据
- 通过 `search_path` 自动切换租户上下文

### 2.2 数据库 Schema 定义

### 2.3 租户上下文切换

```typescript
// tenant-context.ts
import { Pool, PoolClient } from 'pg';

export class TenantContext {
  constructor(private pool: Pool) {}

  async withTenant<T>(tenantId: string, callback: (client: PoolClient) => Promise<T>): Promise<T> {
    const client = await this.pool.connect();
    try {
      // 验证租户存在且有效
      const tenantCheck = await client.query(
        'SELECT id, status FROM public.tenants WHERE id = $1',
        [tenantId]
      );

      if (tenantCheck.rowCount === 0) {
        throw new Error('Tenant not found');
      }

      if (tenantCheck.rows[0].status !== 'active') {
        throw new Error('Tenant is not active');
      }

      // 切换到租户 schema
      const schemaName = `tenant_${tenantId.replace(/-/g, '_')}`;
      await client.query(`SET search_path TO ${schemaName}, public`);

      return await callback(client);
    } finally {
      client.release();
    }
  }
}
```

### 3.2 JWT 认证中间件

```typescript
// middleware/auth.ts
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

interface JWTPayload {
  sub: string;        // user_id
  tid: string;        // tenant_id
  role: string;       // admin, editor, viewer
  iat: number;
  exp: number;
}

declare global {
  namespace Express {
    interface Request {
      user?: JWTPayload;
      tenantId?: string;
    }
  }
}

export const jwtAuth = (req: Request, res: Response, next: NextFunction) => {
  const authHeader = req.headers.authorization;

  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'Missing or invalid authorization header' });
  }

  const token = authHeader.substring(7);

  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET!) as JWTPayload;
    req.user = payload;
    req.tenantId = payload.tid;
    next();
  } catch (err) {
    return res.status(401).json({ error: 'Invalid or expired token' });
  }
};

// API Key 认证
export const apiKeyAuth = async (req: Request, res: Response, next: NextFunction) => {
  const apiKey = req.headers['x-api-key'] as string;

  if (!apiKey) {
    return res.status(401).json({ error: 'Missing API key' });
  }

  // 查询数据库验证 API Key
  const keyRecord = await validateApiKey(apiKey);

  if (!keyRecord) {
    return res.status(401).json({ error: 'Invalid API key' });
  }

  req.tenantId = keyRecord.tenant_id;
  req.apiKeyPermissions = keyRecord.permissions;

  // 更新最后使用时间
  await updateApiKeyLastUsed(keyRecord.id);

  next();
};
```

### 3.3 速率限制实现

```typescript
// middleware/rate-limit.ts
import Redis from 'ioredis';
import { Request, Response, NextFunction } from 'express';

const redis = new Redis(process.env.REDIS_URL);

export const rateLimit = (options: {
  windowMs: number;
  maxRequests: number;
  keyPrefix?: string;
}) => {
  return async (req: Request, res: Response, next: NextFunction) => {
    const key = `${options.keyPrefix || 'rate_limit'}:${req.tenantId || req.ip}`;
    const windowSeconds = Math.ceil(options.windowMs / 1000);

    const current = await redis.incr(key);

    if (current === 1) {
      await redis.expire(key, windowSeconds);
    }

    // 设置响应头
    res.setHeader('X-RateLimit-Limit', options.maxRequests);
    res.setHeader('X-RateLimit-Remaining', Math.max(0, options.maxRequests - current));

    if (current > options.maxRequests) {
      return res.status(429).json({
        error: 'Rate limit exceeded',
        retry_after: await redis.ttl(key)
      });
    }

    next();
  };
};

// 动态速率限制 (从租户配额读取)
export const dynamicRateLimit = async (req: Request, res: Response, next: NextFunction) => {
  const tenantId = req.tenantId;
  if (!tenantId) return next();

  const quota = await getTenantQuota(tenantId);
  const limiter = rateLimit({
    windowMs: 60 * 1000,  // 1分钟
    maxRequests: quota.api_calls_per_min || 100,
    keyPrefix: 'api_limit'
  });

  return limiter(req, res, next);
};
```

### 4.2 任务处理器

```typescript
// queue/processors/content-processor.ts
import { contentQueue } from '../config';
import { ContentPipeline } from '../../services/ContentPipeline';
import { TenantContext } from '../../db/tenant-context';

interface ContentJobData {
  tenantId: string;
  contentId: string;
  stage: 'outline' | 'writing' | 'review';
  prompt: string;
  options?: {
    model?: string;
    temperature?: number;
  };
}

contentQueue.process('generate-outline', 3, async (job) => {
  const { tenantId, contentId, prompt } = job.data as ContentJobData;

  await job.progress(10);

  const pipeline = new ContentPipeline({
    tenantId,
    fallbackEnabled: true
  });

  try {
    // 生成大纲
    await job.progress(30);
    const outline = await pipeline.generateOutline(prompt);

    await job.progress(60);
    // 保存到数据库
    await tenantContext.withTenant(tenantId, async (client) => {
      await client.query(
        `UPDATE contents SET outline = $1, outline_version = outline_version + 1
         WHERE id = $2`,
        [JSON.stringify(outline), contentId]
      );
    });

    await job.progress(100);

    // 触发下一步：写入任务
    await contentQueue.add('generate-content', {
      tenantId,
      contentId,
      stage: 'writing',
      outline
    });

    return { success: true, outline };
  } catch (error) {
    await tenantContext.withTenant(tenantId, async (client) => {
      await client.query(
        `UPDATE contents SET status = 'failed' WHERE id = $1`,
        [contentId]
      );
    });
    throw error;
  }
});

contentQueue.process('generate-content', 5, async (job) => {
  const { tenantId, contentId, outline } = job.data;

  const pipeline = new ContentPipeline({ tenantId });

  await job.progress(20);
  const content = await pipeline.generateContentFromOutline(outline);

  await job.progress(60);
  await tenantContext.withTenant(tenantId, async (client) => {
    await client.query(
      `UPDATE contents SET
        body = $1,
        body_version = body_version + 1,
        word_count = $2,
        reading_time = $3,
        status = 'reviewing'
       WHERE id = $4`,
      [content.body, content.wordCount, content.readingTime, contentId]
    );
  });

  await job.progress(100);

  // 触发蓝军评审
  await reviewQueue.add('blue-team-review', {
    tenantId,
    contentId,
    content: content.body
  });

  return { success: true, wordCount: content.wordCount };
});
```

### 4.3 蓝军评审处理器

```typescript
// queue/processors/review-processor.ts
import { reviewQueue } from '../config';
import { BlueTeamAgent } from '../../services/BlueTeamAgent';

interface ReviewJobData {
  tenantId: string;
  contentId: string;
  content: string;
}

reviewQueue.process('blue-team-review', 2, async (job) => {
  const { tenantId, contentId, content } = job.data as ReviewJobData;

  await job.progress(10);

  const blueTeam = new BlueTeamAgent({
    model: 'claude-sonnet-4-6',
    methodology: '3x3x2'
  });

  // Phase 1: 3 专家独立评审
  await job.progress(30);
  const expertReviews = await blueTeam.runExpertPhase(content);

  // Phase 2: 3 维度深度分析
  await job.progress(60);
  const dimensionReviews = await blueTeam.runDimensionPhase(content, expertReviews);

  // Phase 3: 2 轮综合迭代
  await job.progress(90);
  const finalReview = await blueTeam.runSynthesisPhase(dimensionReviews);

  await job.progress(100);

  // 保存评审结果
  await tenantContext.withTenant(tenantId, async (client) => {
    // 保存详细评审记录
    for (const review of [...expertReviews, ...dimensionReviews]) {
      await client.query(
        `INSERT INTO blue_team_reviews
         (content_id, review_type, expert_role, score, feedback)
         VALUES ($1, $2, $3, $4, $5)`,
        [contentId, review.type, review.role, review.score, review.feedback]
      );
    }

    // 更新内容总评分
    await client.query(
      `UPDATE contents SET
        blue_team_score = $1,
        blue_team_feedback = $2,
        status = CASE WHEN $1 >= 80 THEN 'approved' ELSE 'draft' END
       WHERE id = $3`,
      [finalReview.score, JSON.stringify(finalReview), contentId]
    );
  });

  // 发送通知
  if (finalReview.score < 80) {
    await notifyQueue.add('review-failed', {
      tenantId,
      contentId,
      score: finalReview.score,
      issues: finalReview.criticalIssues
    });
  }

  return finalReview;
});
```

## 5. 从 v2.0 迁移到 v3.0

### 5.1 迁移策略

采用 **Blue-Green 部署** 策略：
1. 保持 v2.0 服务运行
2. 部署 v3.0 到新环境
3. 数据双写：v2.0 写入同时同步到 v3.0
4. 验证 v3.0 功能完整
5. 切换流量到 v3.0
6. 保留 v2.0 一段时间作为 fallback

### 5.2 数据迁移脚本

```typescript
// migrations/v2-to-v3.ts
import { Pool } from 'pg';

const v2Pool = new Pool({ connectionString: process.env.V2_DATABASE_URL });
const v3Pool = new Pool({ connectionString: process.env.V3_DATABASE_URL });

async function migrateTenant(userId: string, tenantSlug: string) {
  // 1. 创建租户
  const tenantResult = await v3Pool.query(
    `INSERT INTO public.tenants (slug, name, plan, created_at)
     VALUES ($1, $2, 'pro', NOW()) RETURNING id`,
    [tenantSlug, `${tenantSlug} Team`]
  );
  const tenantId = tenantResult.rows[0].id;

  // 2. 创建租户 schema
  await v3Pool.query(`CREATE SCHEMA IF NOT EXISTS tenant_${tenantId.replace(/-/g, '_')}`);

  // 3. 迁移 v2.0 数据
  const v2Contents = await v2Pool.query(
    'SELECT * FROM contents WHERE user_id = $1',
    [userId]
  );

  for (const content of v2Contents.rows) {
    await v3Pool.query(
      `INSERT INTO tenant_${tenantId.replace(/-/g, '_')}.contents
       (id, title, body, status, created_at, updated_at, published_at)
       VALUES ($1, $2, $3, $4, $5, $6, $7)`,
      [content.id, content.title, content.body, content.status,
       content.created_at, content.updated_at, content.published_at]
    );
  }

  console.log(`Migrated ${v2Contents.rowCount} contents for tenant ${tenantSlug}`);
}
```

### 5.3 API 兼容性层

```typescript
// compatibility/v2-adapter.ts
// 保持 v2.0 API 格式，内部转发到 v3.0

app.post('/api/v2/content/generate', async (req, res) => {
  // 将 v2 请求转换为 v3 格式
  const v3Request = {
    topic: {
      title: req.body.prompt?.substring(0, 100),
      description: req.body.prompt
    },
    options: {
      model: req.body.model || 'claude-sonnet-4-6'
    }
  };

  // 调用 v3 服务
  const result = await fetch('http://content-pipeline:3000/api/v3/topics', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': req.headers.authorization,
      'X-Tenant-Id': req.headers['x-tenant-id'] || 'legacy'
    },
    body: JSON.stringify(v3Request)
  });

  // 将 v3 响应转回 v2 格式
  const v3Data = await result.json();
  const v2Response = {
    content: v3Data.content?.body,
    model: v3Data.content?.model,
    status: v3Data.content?.status === 'published' ? 'success' : 'pending'
  };

  res.json(v2Response);
});
```

## 8. 关键决策记录 (ADR)

### ADR-001: Schema-per-Tenant vs Row-level 隔离

**决策**: 采用 Schema-per-Tenant

**原因**:
- 数据隔离性强，符合 SaaS 安全要求
- 便于单个租户数据导出/备份
- 查询无需添加 tenant_id 过滤条件

**权衡**:
- 连接池需要管理 schema 切换
- 单数据库租户数上限约 100 (schema 数量限制)

### ADR-002: Bull vs BullMQ vs node-queue

**决策**: 使用 Bull (v4.x)

**原因**:
- 成熟的 Redis-based 队列
- 内置 Dashboard 支持
- 支持延迟任务、优先级、重试策略

### ADR-003: Kong vs 自建 Gateway

**决策**: 使用 Kong

**原因**:
- 开源、轻量
- 内置 JWT、rate-limiting 插件
- 声明式配置便于版本控制
